95 
96  if constexpr(has_recursive_cases)
97  {

99  auto adapted_bco = [bco = static_cast<BCO&&>(*this)]
100  (auto, auto&&... xs) mutable SCELTA_NOEXCEPT_AND_TRT(std::declval<BCO&&>()(FWD(xs)...)) {
101  return bco(FWD(xs)...);
102  };

104  auto o = overload(std::move(adapted_bco), FWD(xs)...);
105  return [rv = make_unresolved_visitor(std::move(o))](auto&&... vs) mutable
106  {
107  // TODO: SFINAE out the conditional because result_of might create instantiation error
108  using rt = std::conditional_t<

112  std::result_of_t<BCO&(original_decay_first_alternative_t<decltype(vs)>...)>,
